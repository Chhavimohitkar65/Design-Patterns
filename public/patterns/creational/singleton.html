Theme Park Ticket System

What is Singleton Pattern?

The Singleton pattern ensures that a class has only one instance and provides a global point of access to that instance. In this example, we demonstrate it through a theme park ticket system where:

  • All ticket windows must share the same ticket counter
  • Total available tickets must be synchronized across all windows
  • Any ticket sale must be reflected immediately at all windows

class TicketSystem {
    private static instance: TicketSystem | null = null;
    private availableTickets: number = 0;
    
    private constructor() {
        // Private constructor prevents direct instantiation
        console.log("Initializing Ticket System");
    }
    
    public static getInstance(): TicketSystem {
        if (!TicketSystem.instance) {
            TicketSystem.instance = new TicketSystem();
        }
        return TicketSystem.instance;
    }
    
    public sellTicket(windowId: string): boolean {
        if (this.availableTickets > 0) {
            this.availableTickets--;
            console.log(`Ticket sold at ${windowId}`);
            return true;
        }
        return false;
    }
    
    public addTickets(count: number): void {
        this.availableTickets += count;
        console.log(`Added ${count} tickets`);
    }
}

// Usage
const window1 = TicketSystem.getInstance();
const window2 = TicketSystem.getInstance();

// Both windows reference the same instance
console.log(window1 === window2); // true
