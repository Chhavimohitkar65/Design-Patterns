Composite Pattern: File System Example

Understanding the Composite Pattern

The Composite pattern allows you to create tree-like structures where both individual objects (files) and compositions of objects (folders) are treated uniformly. Key concepts:

  • Component: The base interface (FileSystemItem)
  • Leaf: Individual objects that can't have children (Files)
  • Composite: Objects that can contain other objects (Folders)

// Component interface
interface FileSystemItem {
    getName(): string;
    getSize(): number;
    getPath(): string;
}

// Leaf class
class File implements FileSystemItem {
    constructor(private name: string, private size: number) {}
    
    getName(): string { return this.name; }
    getSize(): number { return this.size; }
    getPath(): string { return this.name; }
}

// Composite class
class Folder implements FileSystemItem {
    private children: FileSystemItem[] = [];
    
    constructor(private name: string) {}
    
    add(item: FileSystemItem): void {
        this.children.push(item);
    }
    
    getName(): string { return this.name; }
    
    getSize(): number {
        return this.children.reduce(
            (sum, item) => sum + item.getSize(), 
            0
        );
    }
    
    getPath(): string {
        return this.name + '/';
    }
}

// Usage
const root = new Folder('root');
const docs = new Folder('documents');
const file1 = new File('report.pdf', 1024);

root.add(docs);
docs.add(file1);

console.log(root.getSize()); // 1024 (sum of all children)
